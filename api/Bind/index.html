Bind

Allows you to add modifiers to a set of inputs.

local actions = Actions.new({ "look" }):setRebuildBindings(function(bindings)
	bindings:bind(
		"look",
		Bind.new(Enum.KeyCode.Thumbstick2):addModifiers(scale(0.5, Vector2.new(0.5, 0.5)))
	)
end)
